\documentclass[tikz,border=5pt]{standalone}
\usepackage{tikz}
\usepackage{musixtex}
\usetikzlibrary{calc}

\def\distanceScale{0.7} % distance from outer circle to key signature

% Boolean flag for rotating key signatures
\newif\ifRotateKeysigs
%\RotateKeysigstrue   % default: rotated
\RotateKeysigsfalse % uncomment to keep upright

\newsavebox{\MusicBox}

% Create a small pentagram and store in \MusicBox
\newcommand{\keysig}[1]{%
  \begingroup
  \resizebox{!}{12mm}{%
    \begin{music}
      \setclefsymbol1{\hspace{0.3em}}
      \generalsignature{#1}
      \startextract
      \endextract
    \end{music}%
  }%
  \endgroup
}

% Draw the wheel
\newcommand{\drawCircularSectors}[3]{%
    \pgfmathsetmacro{\n}{0}%
    \foreach \x/\y in {#3} {\pgfmathparse{\n+1}\global\let\n=\pgfmathresult}%
    \pgfmathsetmacro{\anglestep}{360/\n}%
    \pgfmathsetmacro{\offset}{\anglestep/2}%

    \foreach \note/\col/\sig [count=\j] in {#3} {
        \pgfmathsetmacro{\startangle}{90-(\j-1)*\anglestep+\offset}%
        \pgfmathsetmacro{\endangle}{\startangle-\anglestep}%
        \pgfmathsetmacro{\midangle}{\startangle-\anglestep/2}%
        \pgfmathsetmacro{\midradius}{(#1+#2)/2}%

        % Draw sector
        \fill[\col]
            (\startangle:#1) arc (\startangle:\endangle:#1) --
            (\endangle:#2) arc (\endangle:\startangle:#2) -- cycle;

        % Sector lines
        \draw[thick] (\startangle:#1) -- (\startangle:#2);
        \draw[thick] (\endangle:#1) -- (\endangle:#2);

        % Note label
        \ifRotateKeysigs
          \node[rotate=\midangle-90, font=\bfseries\large] at (\midangle:\midradius) {\note};
        \else
          \node[font=\bfseries\large] at (\midangle:\midradius) {\note};
        \fi

        % Place pentagram outside wheel
        \ifRotateKeysigs
          \node[anchor=center, inner sep=0pt, outer sep=0pt, rotate=\midangle-90, transform shape] at (\midangle:{#2+\distanceScale}) {\keysig{\sig}};
        \else
          \node[anchor=center, inner sep=0pt, outer sep=0pt] at (\midangle:{#2+\distanceScale}) {\keysig{\sig}};
        \fi
    }

    % Draw circles
    \draw[thick] (0,0) circle (#2);
    \draw[thick] (0,0) circle (#1);
}
