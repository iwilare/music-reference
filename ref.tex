% Create a small pentagram and store in \MusicBox
\newcommand{\keysig}[1]{%
  \begingroup
  \setlength{\unitlength}{1mm}%
  \resizebox{!}{6mm}{%
    \begin{music}
      \parindent0pt
      \smallmusicsize
      \nobarnumbers
      \instrumentnumber{1}
      \setstaffs{1}{1}
      \setclef{1}{\treble}
      \generalsignature{#1}
      \startextract
        \Notes\en
      \endextract
    \end{music}%
  }%
  \endgroup
}

% Draw the wheel
\newcommand{\drawCircularSectors}[3]{%
    \pgfmathsetmacro{\n}{0}%
    \foreach \x/\y in {#3} {\pgfmathparse{\n+1}\global\let\n=\pgfmathresult}%
    \pgfmathsetmacro{\anglestep}{360/\n}%
    \pgfmathsetmacro{\offset}{\anglestep/2}%

    \foreach \note/\col/\sig [count=\j] in {#3} {
        \pgfmathsetmacro{\startangle}{90-(\j-1)*\anglestep+\offset}%
        \pgfmathsetmacro{\endangle}{\startangle-\anglestep}%
        \pgfmathsetmacro{\midangle}{\startangle-\anglestep/2}%
        \pgfmathsetmacro{\midradius}{(2+4)/2}%

        % Draw sector
        \fill[\col]
            (\startangle:2) arc (\startangle:\endangle:2) --
            (\endangle:4) arc (\endangle:\startangle:4) -- cycle;

        % Sector lines
        \draw[thick] (\startangle:2) -- (\startangle:4);
        \draw[thick] (\endangle:2) -- (\endangle:4);

        % Note label
        \node[rotate=\midangle-90, font=\bfseries\large] at (\midangle:\midradius) {\note};

        % Place pentagram outside wheel
        \ifRotateKeysigs
          \node[anchor=center, rotate=\midangle-90] at (\midangle:4.6) {\keysig{\sig}};
        \else
          \node[anchor=center] at (\midangle:4.6) {\keysig{\sig}};
        \fi
    }

    % Draw circles
    \draw[thick] (0,0) circle (4);
    \draw[thick] (0,0) circle (2);
}
